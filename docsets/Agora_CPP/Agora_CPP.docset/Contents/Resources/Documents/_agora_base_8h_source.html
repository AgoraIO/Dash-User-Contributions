36 template<class T>
37 class AutoPtr {
38  typedef T value_type;
39  typedef T* pointer_type;
40 public:
41  AutoPtr(pointer_type p=0)
42  :ptr_(p)
43  {}
44  ~AutoPtr() {
45  if (ptr_)
46  ptr_->release();
47  }
48  operator bool() const { return ptr_ != (pointer_type)0; }
49  value_type& operator*() const {
50  return *get();
51  }
52 
53  pointer_type operator->() const {
54  return get();
55  }
56 
57  pointer_type get() const {
58  return ptr_;
59  }
60 
61  pointer_type release() {
62  pointer_type tmp = ptr_;
63  ptr_ = 0;
64  return tmp;
65  }
66 
67  void reset(pointer_type ptr = 0) {
68  if (ptr != ptr_ && ptr_)
69  ptr_->release();
70  ptr_ = ptr;
71  }
72  template<class C1, class C2>
73  bool queryInterface(C1* c, C2 iid) {
74  pointer_type p = NULL;
75  if (c && !c->queryInterface(iid, (void**)&p))
76  {
77  reset(p);
78  }
79  return p != NULL;
80  }
81 private:
82  AutoPtr(const AutoPtr&);
83  AutoPtr& operator=(const AutoPtr&);
84 private:
85  pointer_type ptr_;
86 };
